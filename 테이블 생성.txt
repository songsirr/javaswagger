create table users(
	user_ID varchar2(20) NOT NULL primary key,
	user_Password varchar2(20) NOT NULL,
	user_Email varchar2(20) NOT NULL UNIQUE,
	user_Phone varchar2(40) NOT NULL UNIQUE
);

create table post(
	post_No number primary key,
	user_Id varchar2(20) references  users(user_id) NOT NULL,
	post_Time date NOT NULL,
	post_DelTime date NOT NULL,
	post_Content varchar2(200) NOT NULL,
	post_Fname varchar2(50)
	
);

create table comments(
	comment_no number primary key,
	post_no number NOT NULL  references post(post_no),
	comment_content varchar2(100) NOT NULL,
	comment_Date Date NOT NULL,
	user_id Varchar2(20) references users(user_id)
);

create table timeline(
	timeLine_No number primary key,
	post_no Number NOT NULL references post(post_no)
);

create table likes(
	post_no number references post(post_no),
	comment_no number references comments(comment_no),
	user_id varchar2(20) references users(user_id)
);

create table follow(
	user_id varchar2(20) references users(user_id),
	follower_id varchar2(20)
);
